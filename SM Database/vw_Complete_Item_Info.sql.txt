CREATE OR REPLACE VIEW 
vw_Complete_Item_Info AS 
        SELECT 
        ITM.CODIGO_SM CODIGO_SM, ITM.NOMBRE NOMBRE, ITM.DESCRIPCION DESCRIPCION, ITM.RUTA RUTA,
        PH.PERIODO PERIODO, PH.PERIODO_DESCRIPCION PERIODO_DESCRIPCION, PH.PERIODO_INFO PERIODO_INFO,
        P.PROCEDENCIA PROCEDENCIA, P.PROCEDENCIA_DESCRIPCION PROCEDENCIA_DESCRIPCION, P.PROCEDENCIA_INFO PROCEDENCIA_INFO,
        M1.MATERIAL MATERIAL_1, M1.MATERIAL_INFO MATERIAL_INFO_1,
        M2.MATERIAL MATERIAL_2, M2.MATERIAL_INFO MATERIAL_INFO_2,
        UF1.USO USO_1, UF1.USO_INFO USO_INFO_1,
        UF2.USO USO_2, UF2.USO_INFO USO_INFO_2,
        CD.NOMBRE CEDULA_NOMBRE, CD.PROCEDENCIA CEDULA_PROCEDENCIA, CD.PERIODO CEDULA_PERIODO, CD.REFERENCIA_MUNAE CEDULA_REFERENCIA_MUNAE,
        M.MAPA MAPA,
        UNESCO.CLASIFICACION UNESCO_CLASIFICACION, UNESCO.CLASIFICACION_INFO UNESCO_CLASIFICACION_INFO,
        I.ID_IDIOMA IDIOMA
        FROM SM_ITEM ITM
          INNER JOIN SM_PERIODOHISTORICO PH
            ON ITM.ID_PERIODOHISTORICO = PH.ID_PERIODOHISTORICO
          INNER JOIN SM_PROCEDENCIA P
            ON ITM.ID_PROCEDENCIA = P.ID_PROCEDENCIA
          INNER JOIN SM_MATERIAL M1
            ON ITM.ID_MATERIAL_1 = M1.ID_MATERIAL  
          LEFT JOIN SM_MATERIAL M2
            ON ITM.ID_MATERIAL_2 = M2.ID_MATERIAL
          INNER JOIN SM_USOYFORMA UF1
            ON ITM.ID_USOYFORMA_1 = UF1.ID_USOYFORMA
          LEFT JOIN SM_USOYFORMA UF2
            ON ITM.ID_USOYFORMA_2 = UF2.ID_USOYFORMA
          INNER JOIN SM_CEDULA CD
            ON ITM.ID_CEDULA = CD.ID_CEDULA
          LEFT JOIN SM_MAPA M
            ON ITM.ID_MAPA = M.ID_MAPA
          LEFT JOIN SM_CLASIFICACIONUNESCO UNESCO
            ON ITM.ID_CLASIFICACIONUNESCO = UNESCO.ID_CLASIFICACIONUNESCO
          INNER JOIN SM_IDIOMA I
            ON ITM.ID_IDIOMA = I.ID_IDIOMA
